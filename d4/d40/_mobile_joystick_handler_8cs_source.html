21 public void OnDrag(PointerEventData _eventData)
22 {
23 // Calculate the half size difference between the background and handle rects
24 Vector2 difference = new Vector2
25 {
26 x = (background.rect.size.x - handle.rect.size.x) * .5f,
27 y = (background.rect.size.y - handle.rect.size.y) * .5f
28 };
29
30 // Calculate the axis of the input based on the event data and the relative
31 // position to the background's center
32 Axis = Vector2.ClampMagnitude(new Vector2
33 {
34 x = (_eventData.position.x - background.position.x) / difference.x,
35 y = (_eventData.position.y - background.position.y) / difference.y
36 }, 1);
37
38 // Apply the axis position to the handles position
39 handle.position = new Vector3
40 {
41 x = (Axis.x * difference.x) + background.position.x,
42 y = (Axis.y * difference.y) + background.position.y
43 };
44
45 // Apply the deadzone effect after the handle has been placed to prevent
46 // the handle from visually being stuck in the deadzone
47 Axis = (Axis.magnitude < deadzone) ? Vector2.zero : Axis;
48 }
